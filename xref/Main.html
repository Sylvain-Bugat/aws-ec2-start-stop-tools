1   import java.net.URL;
2   import java.security.ProtectionDomain;
3   
4   import org.eclipse.jetty.server.NetworkTrafficServerConnector;
5   import org.eclipse.jetty.server.Server;
6   import org.eclipse.jetty.webapp.Configuration.ClassList;
7   import org.eclipse.jetty.webapp.WebAppContext;
8   
9   /**
10   * Jetty main launcher class.
11   * 
12   * @author Sylvain Bugat
13   */
14  public class Main {
15  
16  	/**
17  	 * Launcher and program main method.
18  	 * 
19  	 * @param args program arguments
20  	 * @throws Exception thrown by Jetty
21  	 */
22  	public static void main(final String[] args) throws Exception {
23  
24  		// Jetty server
25  		final Server server = new Server();
26  
27  		// Add Jetty Web XML and annotation configuration
28  		final ClassList classList = ClassList.setServerDefault(server);
29  		classList.addBefore("org.eclipse.jetty.webapp.JettyWebXmlConfiguration", "org.eclipse.jetty.annotations.AnnotationConfiguration"); //$NON-NLS-1$ //$NON-NLS-2$
30  
31  		// Add a connector to open a port
32  		try (final NetworkTrafficServerConnector connector = new NetworkTrafficServerConnector(server)) {
33  
34  			connector.setPort(8080);
35  			server.addConnector(connector);
36  
37  			// Get the current class URL to add the current packaged war
38  			final ProtectionDomain mainDomain = Main.class.getProtectionDomain();
39  			final URL location = mainDomain.getCodeSource().getLocation();
40  
41  			// Add the war location to the context, the default context path is /
42  			final WebAppContext webAppContext = new WebAppContext();
43  			webAppContext.setWar(location.toExternalForm());
44  			server.setHandler(webAppContext);
45  
46  			// Start the Jetty server
47  			server.start();
48  			server.join();
49  		}
50  	}
51  }
